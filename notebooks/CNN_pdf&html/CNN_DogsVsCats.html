def build_cnn(nb_conv):
    model = Sequential()

    # 1re convolution
    model.add(Conv2D(32, (3,3), activation='relu', input_shape=(64,64,3)))
    model.add(MaxPooling2D((2,2)))

    # 2me convolution
    model.add(Conv2D(32, (3,3), activation='relu'))
    model.add(MaxPooling2D((2,2)))

    # 3me convolution OPTIONNELLE
    if nb_conv == 3:
        model.add(Conv2D(64, (3,3), activation='relu'))
        model.add(MaxPooling2D((2,2)))

    model.add(Flatten())
    model.add(Dense(128, activation='relu'))
    model.add(Dense(1, activation='sigmoid'))

    model.compile(
        optimizer='adam',
        loss='binary_crossentropy',
        metrics=['accuracy']
    )

    return model

def get_callbacks(nb_conv):
    ckpt_dir = f"checkpoints_cnn_{nb_conv}"
    os.makedirs(ckpt_dir, exist_ok=True)

    checkpoint = ModelCheckpoint(
        filepath=os.path.join(ckpt_dir, "weights_epoch_{epoch:02d}.h5"),
        monitor="val_loss",
        save_weights_only=True,
        save_best_only=False,
        verbose=1
    )

    early_stop = EarlyStopping(
        monitor="val_loss",
        patience=3,
        restore_best_weights=True,
        verbose=1
    )

    return checkpoint, early_stop, ckpt_dir

Le graphique comparatif montre que CNN 3 convolutions converge lgrement plus rapidement et atteint une prcision suprieure, ce qui confirme que la couche convolutionnelle supplmentaire permet au modle de mieux capturer les caractristiques des images, mme pour une tche binaire comme Dogs & Cats.

Donc l'ajout de la troisime convolution amliore la performance globale du CNN, sans provoquer doverfitting significatif. Pour ce dataset, le CNN 3 convolutions est donc lgrement prfrable.
